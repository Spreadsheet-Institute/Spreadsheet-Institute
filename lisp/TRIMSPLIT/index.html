TRIMSPLIT



=TRIMSPLIT("   a , b  ,    c    ", ",")
-> {"a","b","c"}
        


DEFINE(
    TRIMSPLIT,
    LAMBDA(
        source_text,
        split_at,
        [display_vertically],
        LET(
            result,
                TRIMALL(
                    TEXTSPLIT(
                        source_text,
                        split_at)),
            IF(
                DEFAULT(display_vertically, FALSE),
                TRANSPOSE(result),
                result))))
