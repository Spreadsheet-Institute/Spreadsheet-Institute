DEFINE(
    IFBLANK,
    LAMBDA(
        cell_reference,
        value_if_blank,
        value_if_not_blank,
        IF(
            OR(
                ISBLANK(cell_reference),
                EQUAL(
                    cell_reference,
                    ""),
            value_if_blank,
            value_if_not_blank))))
