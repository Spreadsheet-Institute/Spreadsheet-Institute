CELLNAME



=CELLNAME(A1)
-> "A1"

=CELLNAME(A1, YES)
-> "$A$1"
        


DEFINE(
    CELLNAME,
    LAMBDA(
        [reference],
        [absolute?],
        LET(
            address,
                IFOMITTED(
                    reference,
                    CELL("address"),
                    CELL(
                        "address",
                        reference)),
            display_absolute,
                DEFAULT(
                    absolute?,
                    NO),
            IF(
                display_absolute,
                address,
                SUBSTITUTE(
                    address,
                    "$",
                    "")))))
