DEFINE(
    SELECTFROM,
    LAMBDA(
        columns,
        table_range,
        [row_conditions],
        LET(
            row_subset,
                SELECTROWS(
                    table_range,
                    row_conditions),
            SELECTCOLUMNS(
                row_subset,
                columns))))
