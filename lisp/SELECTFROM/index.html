DEFINE(
    SELECTFROM,
    LAMBDA(
        columns,
        table_range,
        [row_conditions],
        LET(
            column_names,
                IF(
                    RANGE?(columns),
                    columns,
                    TRIMSPLIT(
                        columns,
                        ",")),
            row_subset,
                SELECTROWS(
                    table_range,
                    row_conditions),
            column_indices,
                INDICES(
                    column_names,
                    FIRSTROW(table_range)),
            SELECTCOLUMNS(
                row_subset,
                column_indices))))
