DEFINE(
    SUMFROM,
    LAMBDA(
        column,
        table_range,
        [row_conditions],
        SUM(
            HEADLESS(
                SELECTCOLUMNS(
                    SELECTROWS(
                        table_range,
                        row_conditions),
                    columns)))))
