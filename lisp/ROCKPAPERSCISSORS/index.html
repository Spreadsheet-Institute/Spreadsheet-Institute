    =ROCKPAPERSCISSORS("Paper")
    -> "You threw PAPER. SL threw ROCK. "

        
            
    DEFINE(
        ROCKPAPERSCISSORS,
        LAMBDA(
            throw,
            LET(
                human,
                    UPPER(throw),
                robot,
                    CHOOSE(
                        RANDBETWEEN(1, 3),
                        "ROCK",
                        "PAPER",
                        "SCISSORS"),
                IFS(
                    EQUAL(human, robot),
                        FORMAT("Tie! {1} vs. {1}", human),
                    AND(EQUAL(human, "PAPER"), EQUAL(robot, "ROCK")),
                        "You win! Paper covers Rock.",
                    AND(EQUAL(human, "PAPER"), EQUAL(robot, "SCISSORS")),
                        "You lose! Scissors cuts Paper.",
                    AND(EQUAL(human, "ROCK"), EQUAL(robot, "SCISSORS")),
                        "You win! Rock crushes Scissors.",
                    AND(EQUAL(human, "ROCK"), EQUAL(robot, "PAPER")),
                        "You lose! Paper covers Rock.",
                    AND(EQUAL(human, "SCISSORS"), EQUAL(robot, "PAPER")),
                        "You win! Scissors cuts Paper.",
                    AND(EQUAL(human, "SCISSORS"), EQUAL(robot, "ROCK")),
                        "You lose! Rock crushes Scissors.",
                    OTHERWISE,
                        "Invalid throw. Please choose Rock, Paper, or Scissors."))))
