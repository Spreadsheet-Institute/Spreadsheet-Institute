DEFINE(
    SELECTROWS,
    LAMBDA(
        range,
        [row_conditions],
        IFOMITTED(
            row_conditions,
            range,
            LET(
                column_names,
                    FIRSTROW(range),
                condition_strings,
                    TRIMSPLIT(
                        row_conditions,
                        ",",
                        YES),
                condition_count,
                    COUNTA(condition_strings),
                condition_operators,
                    EXTRACTOPERATORS(condition_strings),
                condition_columns,
                    MAKEARRAY(
                        condition_count,
                        1,
                        LAMBDA(
                            row,
                            _col,
                            LET(
                                condition,
                                    INDEX(
                                        condition_strings,
                                        row,
                                        1),
                                operator,
                                    INDEX(
                                        condition_operators,
                                        row,
                                        1),
                                TRIM(
                                    TEXTBEFORE(
                                        condition,
                                        operator))))),
                condition_criteria,
                    MAKEARRAY(
                        condition_count,
                        1,
                        LAMBDA(
                            row,
                            _col,
                            LET(
                                condition,
                                    INDEX(
                                        condition_strings,
                                        row,
                                        1),
                                operator,
                                    INDEX(
                                        condition_operators,
                                        row,
                                        1),
                                result,
                                    TRIM(
                                        TEXTAFTER(
                                            condition,
                                            operator)),
                                IFERROR(
                                    NUMBERVALUE(result),
                                    result)))),
                filter_lookup_table,
                    HSTACK(
                        condition_operators,
                        condition_columns,
                        condition_criteria),
                VSTACK(
                    FIRSTROW(range),
                    SIEVE(
                        TABLEDATA(range),
                        filter_lookup_table))))))
