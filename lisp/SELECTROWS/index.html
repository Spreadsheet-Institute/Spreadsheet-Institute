SELECTROWS



=SELECTROWS(A1:Z100, "Salary > 100000, Department = IT")
-> A1:Z10
        


DEFINE(
    SELECTROWS,
    LAMBDA(
        range,
        [row_conditions],
        IFOMITTED(
            row_conditions,
            range,
            LET(
                column_names,
                    FIRSTROW(range),
                condition_strings,
                    TRIMSPLIT(
                        row_conditions,
                        ","),
                condition_count,
                    COUNTA(condition_strings),
                OPERATOR,
                    LAMBDA(
                        condition,
                        IFS(
                            FIND("<=", condition),
                                HSTACK("<=", LTE),
                            FIND(">=", condition),
                                HSTACK(">=", GTE),
                            FIND("<", condition),
                                HSTACK("<", LESSTHAN),
                            FIND(">", condition),
                                HSTACK(">", GREATERTHAN),
                            FIND("=", condition),
                                HSTACK("=", EQUAL),
                            OTHERWISE,
                                NA())),
                condition_columns,
                    MAKEARRAY(
                        condition_count,
                        1,
                        LAMBDA(
                            row,
                            _col,
                            TEXTBEFORE(MATCH(condition_strings, row, 1)))),
                filter_functions,
                    MAKEARRAY(
                        condition_count,
                        1,
                        LAMBDA(
                            row,
                            _col,
                            OPERATOR(TEXTBETWEEN(MATCH(list_of_condition_strings, row, 1), " ", " ")))),
                filter_thresholds,
                    MAKEARRAY(
                        condition_count,
                        1,
                        LAMBDA(
                            row,
                            _col,
                            TEXTAFTER(MATCH(list_of_condition_strings, row, 1), " ", 2))),
                filter_table,
                    HSTACK(
                        filter_columns,
                        filter_functions,
                        filter_thresholds),
                MEETSCRITERIA,
                    LAMBDA(
                        row,
                        IF(
                            ALL(
                                MAKEARRAY(
                                    condition_count,
                                    1,
                                    LAMBDA(
                                        row,
                                        _col,
                                        LET(
                                            filter_column,
                                                INDEX(filter_table, row, 1),
                                            filter_function,
                                                INDEX(filter_table, row, 2),
                                            filter_threshold,
                                                INDEX(filter_table, row, 3),
                                            APPLY(filter_function, INDEX(row, 1, MATCH(filter_column, column_names)), filter_threshold))))))),
                VSTACK(
                    FIRSTROW(range),
                    BYROW(
                        range,
                        MEETSCRITERIA()))))))
