DEFINE(
    SELECTROWS,
    LAMBDA(
        table_range,
        [row_conditions],
        IFOMITTED(
            row_conditions,
            table_range,
            LET(
                column_names,
                    FIRSTROW(table_range),
                VSTACK(
                    column_names,
                    SIEVE(
                        HEADLESS(table_range),
                        CRITERIATABLE(
                            column_names,
                            row_conditions)))))))
