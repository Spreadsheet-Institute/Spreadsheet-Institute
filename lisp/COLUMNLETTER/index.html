COLUMNLETTER



=COLUMNLETTER(A1)
-> "A"

=COLUMNLETTER(AB100)
-> "AB"
        


DEFINE(
    COLUMNLETTER,
    LAMBDA(
        [cell_reference],
        TEXTBETWEEN(
            IFOMITTED(
                cell_reference,
                ADDRESS(
                    ROW(),
                    COLUMN()),
                ADDRESS(
                    ROW(cell_reference),
                    COLUMN(cell_reference))),
            "$",
            "$")))
