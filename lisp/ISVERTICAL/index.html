ISVERTICAL



=ISVERTICAL(A1:E1)
-> FALSE

=ISVERTICAL(A1:A5)
-> TRUE
        


DEFINE(
    ISVERTICAL,
    LAMBDA(
        range,
        IF(
            GREATERTHAN(
                ROWS(range),
                COLUMNS(range)),
            TRUE,
            FALSE)))
