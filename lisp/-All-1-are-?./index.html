DEFINE(
    _All_1_are_?.,
    LAMBDA(
        _1,
        knowledgebase,
        LET(
            fragment,
                FORMAT(
                    "All {1} are ",
                    _1),
            knowledgecolumn,
                TOCOL(knowledgebase),
            matches,
                FILTER(
                    knowledgecolumn,
                    BEGINSWITH?(
                        knowledgecolumn,
                        fragment)),
            IF(
                ISERROR(matches),
                FALSE,
                TRIMALL(
                    REGEXEXTRACT(
                        matches,
                        FORMAT(
                            "^All {1} are (.+)\.$",
                            _1),
                        2))))))
