DEFINE(
    _Is_1_2?,
    LAMBDA(
        _1,
        _2,
        knowledgebase,
        LET(
            affirmation,
                CONTAINS(
                    knowledgebase,
                    FORMAT(
                        "{1} is {2}.",
                        CAPITALIZE(_1),
                        _2)),
            negation,
                CONTAINS(
                    knowledgebase,
                    FORMAT(
                        "{1} is not {2}.",
                        CAPITALIZE(_1),
                        _2)),
            template,
                IFS(
                    AND(
                        affirmation,
                        negation),
                        "Yes and no. {1} simultaneously is and is not {2}.",
                    affirmation,
                        "Yes. {1} is {2}.",
                    negation,
                        "No. {1} is not {2}.",
                    OTHERWISE,
                        "Unknown."),
            FORMAT(
                template,
                _1,
                _2))))
