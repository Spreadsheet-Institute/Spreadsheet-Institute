The =BINGO() function



ABSTRACT

The =BINGO([options]) function returns a complete/random game of bingo, which is either 75 standard bingo balls or 24 text options (if provided).



EXAMPLE

A1=BINGO()
->
  A B
1 G 51
... ... ...
75 B 12 


DEFINE(
    BINGO,
    LAMBDA(
        [list_of_options],
        IF(
            PROVIDED?(list_of_options),
            SHUFFLED(list_of_options),
            LET(
                shuffled_bingo_balls,
                    SHUFFLED(
                        SEQUENCE(
                            75,
                            1,
                            1)),
                HSTACK(
                    MAKEARRAY(
                        75,
                        1,
                        LAMBDA(
                            ball_number,
                            _col,
                            IFS(
                                BETWEEN?(ball_number, 1, 15),
                                    "B",
                                BETWEEN?(ball_number, 16, 30),
                                    "I",
                                BETWEEN?(ball_number, 31, 45),
                                    "N",
                                BETWEEN?(ball_number, 46, 60),
                                    "G",
                                OTHERWISE,
                                    "O"))),
                    shuffled_bingo_balls)))))
