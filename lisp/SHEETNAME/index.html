DEFINE(
    SHEETNAME,
    LAMBDA(
        [reference],
        LET(
            filename,
                FILENAME(reference),
            bracket_position,
                FIND(
                    "]",
                    filename),
            RIGHT(
                filename,
                DECREMENT(
                    LEN(filename),
                    bracket_position)))))
