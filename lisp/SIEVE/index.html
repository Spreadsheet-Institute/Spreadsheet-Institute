    SIEVE,
    LAMBDA(
        table_data,
        filter_lookup_table,
        FILTER(
            table_data,
            EQUAL(
                TRUE,
                BYROW(
                    table_data,
                    LAMBDA(
                        row,
                        MEETSCRITERIA(
                            row,
                            filter_lookup_table)))))))
