DEFINE(
    SELECT,
    LAMBDA(
        columns,
        source_range,
        [conditions],
        LET(
            column_names,
                IF(
                    RANGE?(columns),
                    columns,
                    TRIMALL(
                        TEXTSPLIT(
                            columns,
                            ","))),
            column_indices,
                INDICES(
                    column_names,
                    FIRSTROW(source_range)),
            column_subset,
                FILTERCOLUMNS(
                    source_range,
                    column_indices),
            IFOMITTED(
                conditions,
                column_subset,
                FILTERROWS(
                    column_subset,
                    conditions)))))
