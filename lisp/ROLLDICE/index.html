DEFINE(
    ROLLDICE,
    LAMBDA(
        [times],
        IF(
            LTE(
                DEFAULT(times, 1),
                1),
            DICEROLL(),
            CONS(
                DICEROLL(),
                ROLLDICE(
                    DECREMENT(times))))))
