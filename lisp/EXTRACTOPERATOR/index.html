DEFINE(
    EXTRACTOPERATOR,
    LAMBDA(
        condition_string,
        IFS(
            ISNUMBER(FIND("<>", condition_string)),
                HSTACK("<>", NOTEQUAL),
            ISNUMBER(FIND("<=", condition_string)),
                HSTACK("<=", LTE),
            ISNUMBER(FIND(">=", condition_string)),
                HSTACK(">=", GTE),
            ISNUMBER(FIND("<", condition_string)),
                HSTACK("<", LESSTHAN),
            ISNUMBER(FIND(">", condition_string)),
                HSTACK(">", GREATERTHAN),
            ISNUMBER(FIND("!=", condition_string)),
                HSTACK("!=", NOTEQUAL),
            ISNUMBER(FIND("=", condition_string)),
                HSTACK("=", EQUAL),
            OTHERWISE,
                NA())))
