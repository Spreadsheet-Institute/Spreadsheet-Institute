DEFINE(
    _1_is_?.,
    LAMBDA(
        _1,
        knowledgebase,
        LET(
            fragment,
                FORMAT(
                    "{1} is ",
                    _1),
            knowledgecolumn,
                TOCOL(knowledgebase),
            matches,
                FILTER(
                    knowledgecolumn,
                    BEGINSWITH?(
                        knowledgecolumn,
                        fragment)),
            IF(
                ISERROR(matches),
                FALSE,
                TRIMALL(
                    REGEXEXTRACT(
                        matches,
                        FORMAT(
                            "^{1} is (.+)\.$",
                            _1),
                        2))))))
