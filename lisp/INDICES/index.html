INDICES



=INDICES({"a","c"}, {"a","b","c"})
-> {1,3}

=INDICES({"a";"b"}, {"a";"b";"c"})
-> {1,2}
        


DEFINE(
    INDICES,
    LAMBDA(
        subset,
        superset,
        LET(
            horizontal_subset,
                IF(
                    VERTICAL?(subset),
                    TRANSPOSE(subset),
                    subset),
            horizontal_superset,
                IF(
                    VERTICAL?(superset),
                    TRANSPOSE(superset),
                    superset),
            MAKEARRAY(
                1,
                IF(
                    RANGE?(horizontal_subset),
                    LENGTH(horizontal_subset),
                    1),
                LAMBDA(
                    _row,
                    col,
                    MATCH(
                        INDEX(
                            horizontal_subset,
                            1,
                            col),
                        horizontal_superset,
                        FALSE))))))
