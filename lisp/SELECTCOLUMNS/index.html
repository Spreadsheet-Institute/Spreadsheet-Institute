DEFINE(
    SELECTCOLUMNS,
    LAMBDA(
        range,
        column_indices,
        IF(
            ONE?(LENGTH(column_indices)),
            CHOOSECOLS(
                range,
                column_indices),
            HSTACK(
                CHOOSECOLS(
                    range,
                    FIRST(column_indices)),
                SELECTCOLUMNS(
                    range,
                    REST(column_indices))))))
