DEFINE(
    _All_?_are_1.,
    LAMBDA(
        _1,
        knowledgebase,
        LET(
            predicate,
                FORMAT(
                    " are {1}.",
                    _1),
            knowledgecolumn,
                TOCOL(knowledgebase),
            matches,
                FILTER(
                    knowledgecolumn,
                    ENDSWITH?(
                        knowledgecolumn,
                        predicate)),
            IF(
                ISERROR(matches),
                FALSE,
                TRIMALL(
                    REGEXEXTRACT(
                        matches,
                        FORMAT(
                            "^All (.+) are {1}\.$",
                            _1),
                        2))))))
