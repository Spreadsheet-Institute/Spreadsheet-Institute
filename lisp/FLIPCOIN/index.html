DEFINE(
    FLIPCOIN,
    LAMBDA(
        [times],
        LET(
            counter,
                DEFAULT(times, 1),
            IF(
                LTE(counter, 1),
                COINTOSS(),
                VSTACK(
                    COINTOSS(),
                    FLIPCOIN(DECREMENT(counter)))))))
